Var
// Secao de Declaracoes das variaveis 
j: inteiro;

Inicio
// Secao de Comandos, procedimento, funcoes, operadores, etc...  
    escreval("Imprimindo de 1 a 10 repita ate: ")
        j<-1
    repita
        escreval(j)
        j<-j+1
        ate j>10
    escreval("Valor de J: ", j)
